```javascript
// Simplified mass print without custom fonts
app.get('/api/payslips/month/:month/print', authenticateToken, async (req, res) => {
  try {
    const { month } = req.params;
    console.log('Mass print requested for month:', month);
    
    // Your existing payslips query here
    const payslips = await getPayslipsForMonth(month);
    
    if (!payslips || payslips.length === 0) {
      return res.status(404).json({ message: 'No payslips found for this month' });
    }
    
    const PDFDocument = require('pdfkit');
    const doc = new PDFDocument();
    
    res.setHeader('Content-Type', 'application/pdf');
    res.setHeader('Content-Disposition', `attachment; filename="payslips-${month}.pdf"`);
    
    doc.pipe(res);
    doc.font('Helvetica'); // Use default font
    
    // Simple payslip generation
    payslips.forEach((payslip, index) => {
      if (index > 0) doc.addPage();
      
      doc.fontSize(16).text('PAYSLIP', { align: 'center' });
      doc.moveDown();
      doc.fontSize(12);
      doc.text(`Employee: ${payslip.employee_name || 'N/A'}`);
      doc.text(`Month: ${payslip.month || month}`);
      doc.text(`Basic Salary: €${payslip.basic_salary || '0.00'}`);
      doc.text(`Net Salary: €${payslip.net_salary || '0.00'}`);
    });
    
    doc.end();
    
  } catch (error) {
    console.error('Mass print error:', error);
    res.status(500).json({ message: 'PDF generation failed: ' + error.message });
  }
});
```
