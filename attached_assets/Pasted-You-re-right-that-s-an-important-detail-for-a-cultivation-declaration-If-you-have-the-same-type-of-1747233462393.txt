import type { NextApiRequest, NextApiResponse } from 'next';
import { db } from '@/server/db'; // Adjust path to your db setup
import { plantsTable } from '@/server/db/schema'; // Adjust path to your schema
import { asc, gt } from 'drizzle-orm'; // asc is already here, gt for excludeZero
import { PDFDocument, rgb, PageSizes, PDFFont } from 'pdf-lib';
import fs from 'fs';
import path from 'path';
const drawTextSafe = (
page: any,
text: any,
x: number,
y: number,
font: PDFFont,
size: number,
maxWidth: number,
color = rgb(0, 0, 0)
) => {
// ... (keep existing drawTextSafe function from previous step with truncation)
let safeText = text === undefined || text === null ? '' : String(text);
let textWidth = font.widthOfTextAtSize(safeText, size);
if (textWidth > maxWidth && maxWidth > 0) {
let truncatedText = '';
for (let i = safeText.length; i > 0; i--) {
truncatedText = safeText.substring(0, i) + '...';
if (font.widthOfTextAtSize(truncatedText, size) <= maxWidth) {
break;
}
}
safeText = truncatedText;
if (font.widthOfTextAtSize(safeText, size) > maxWidth) {
safeText = safeText.substring(0, Math.floor(maxWidth / font.widthOfTextAtSize("...", size) * 3));
}
}
page.drawText(safeText, { x, y, font, size, color });
};
export default async function handler(
req: NextApiRequest,
res: NextApiResponse
) {
if (req.method === 'GET') {
const { excludeZero } = req.query;
console.log(API: Cultivation Report. Exclude zero: ${excludeZero}. Adding secondary sort by planting year.);
try {
  const fontPath = path.join(process.cwd(), 'public', 'fonts', 'NotoSansGreek-Regular.ttf');
  let customFontBytes;
  try {
    customFontBytes = fs.readFileSync(fontPath);
  } catch (fontError: any) {
    console.error(`API: Failed to load font file at ${fontPath}. Error: ${fontError.message}`);
    return res.status(500).json({ 
      message: 'Failed to load font for PDF generation. Ensure font file is in public/fonts/.', 
      error: `Font file not found: ${fontPath}.` 
    });
  }

  // Base query builder
  let queryBuilder = db.select().from(plantsTable);

  // Conditionally add where clause for excluding zero quantities
  if (excludeZero === 'true') {
    console.log("API: Excluding plants with zero quantity.");
    // @ts-ignore Drizzle's fluent API should handle this chaining.
    queryBuilder = queryBuilder.where(gt(plantsTable.quantity, 0));
  }
  
  // Fetch and sort data: primary sort by name, secondary sort by plantingYear
  // @ts-ignore Drizzle's fluent API should handle this chaining.
  const plantsData = await queryBuilder.orderBy(asc(plantsTable.name), asc(plantsTable.plantingYear));
  
  console.log(`API: Fetched ${plantsData.length} plants for the report (sorted by name, then year).`);

  const pdfDoc = await PDFDocument.create();
  const customFont = await pdfDoc.embedFont(customFontBytes);
  
  let page = pdfDoc.addPage(PageSizes.A4);
  const { width, height } = page.getSize();
  
  const fontSize = 9;
  const headerFontSize = 10;
  const titleFontSize = 14;
  const margin = 40;
  let yPosition = height - margin - 20;

  const reportTitle = "ΔΗΛΩΣΗ ΚΑΛΛΙΕΡΓΕΙΑΣ ΓΙΑ ΤΟ 2025";
  const titleWidth = customFont.widthOfTextAtSize(reportTitle, titleFontSize);
  page.drawText(reportTitle, {
    x: (width - titleWidth) / 2,
    y: yPosition,
    font: customFont,
    size: titleFontSize,
    color: rgb(0, 0, 0),
  });
  yPosition -= (titleFontSize + 25);

  const headers = [
    { text: 'Α/Α', width: 35, xOffset: 5 },
    { text: 'Name', width: 170, xOffset: 5 },
    { text: 'Scientific Name', width: 170, xOffset: 5 },
    { text: 'Year', width: 60, xOffset: 5 },
    { text: 'Quantity', width: 60, xOffset: 5 },
  ];

  const rowHeight = 18;

  const drawPageHeaders = (targetPage: any, currentY: number) => {
    // ... (keep existing drawPageHeaders function)
    let x = margin;
    targetPage.drawLine({ start: { x: margin, y: currentY + 5 }, end: { x: width - margin, y: currentY + 5 }, thickness: 1, color: rgb(0,0,0) });

        x += header.width;
    });
    targetPage.drawLine({ start: { x: margin, y: currentY - rowHeight + 7 }, end: { x: width - margin, y: currentY - rowHeight + 7 }, thickness: 1, color: rgb(0,0,0) });
    return currentY - rowHeight;
  };
  
  yPosition = drawPageHeaders(page, yPosition + rowHeight);

  plantsData.forEach((plant, index) => {
    // ... (keep existing row drawing logic, including new page creation and header redraw)
    if (yPosition < margin + rowHeight) {
      page.drawLine({start: {x: margin, y: yPosition + 7}, end: {x: width-margin, y: yPosition + 7}, thickness: 0.5, color: rgb(0.7,0.7,0.7) });
      page = pdfDoc.addPage(PageSizes.A4);
      yPosition = height - margin - rowHeight - 10; 
      yPosition = drawPageHeaders(page, yPosition + rowHeight);
    }

    let currentX = margin;
    const rowData = [
      { text: (index + 1).toString(), width: headers[0].width, xOffset: headers[0].xOffset },
      { text: plant.name, width: headers[1].width, xOffset: headers[1].xOffset },
      { text: plant.scientificName || '', width: headers[2].width, xOffset: headers[2].xOffset },
      { text: plant.plantingYear?.toString() || '', width: headers[3].width, xOffset: headers[3].xOffset },
      { text: plant.quantity.toString(), width: headers[4].width, xOffset: headers[4].xOffset },
    ];

      const cellContentMaxWidth = data.width - data.xOffset - 2;

      currentX += data.width;
    });
    page.drawLine({start: {x: margin, y: yPosition - rowHeight + 7}, end: {x: width-margin, y: yPosition -rowHeight + 7}, thickness: 0.5, color: rgb(0.7,0.7,0.7)});
    yPosition -= rowHeight;
  });

  const pdfBytes = await pdfDoc.save();
  const timestamp = new Date().toISOString().replace(/:/g, '-').slice(0, 19);
  const filename = `declaration-cultivation-2025-${excludeZero === 'true' ? 'nozero-' : ''}${timestamp}.pdf`;

  res.setHeader('Content-Disposition', `attachment; filename="${filename}"`);
  res.setHeader('Content-Type', 'application/pdf');
  res.status(200).send(Buffer.from(pdfBytes));
  console.log(`API: Cultivation Declaration Report sent. Excluded zero: ${excludeZero}, sorted by name then year.`);

} catch (error: any) {
  console.error('API: Error generating Cultivation Declaration PDF:', error);
  res.status(500).json({ message: 'Failed to generate PDF report', error: error.message });
}

} else {
res.setHeader('Allow', ['GET']);
res.status(405).end(Method ${req.method} Not Allowed);
}
}
