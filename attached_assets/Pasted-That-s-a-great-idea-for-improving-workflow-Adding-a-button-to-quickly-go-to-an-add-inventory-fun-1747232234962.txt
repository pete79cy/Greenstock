import type { NextApiRequest, NextApiResponse } from 'next';
import { db } from '@/server/db'; // Adjust to your DB setup path
import { plantsTable } from '@/server/db/schema'; // Adjust to your schema path
import { eq, sql } from 'drizzle-orm';

export default async function handler(
req: NextApiRequest,
res: NextApiResponse
) {
if (req.method \!== 'POST') {
res.setHeader('Allow', ['POST']);
return res.status(405).end(`Method ${req.method} Not Allowed`);
}

const plantId = req.query.id as string;
const { quantityToAdd } = req.body;

console.log(`API: Attempting to add stock for plant ID: ${plantId}, quantity to add: ${quantityToAdd}`);

if (\!plantId || quantityToAdd === undefined) {
console.log("API: Missing plantId or quantityToAdd");
return res.status(400).json({ message: 'Plant ID and quantity to add are required.' });
}

const id = parseInt(plantId, 10);
const amount = parseInt(quantityToAdd, 10);

if (isNaN(id)) {
console.log(`API: Invalid plant ID format: ${plantId}`);
return res.status(400).json({ message: 'Invalid Plant ID format.' });
}

if (isNaN(amount) || amount \<= 0) {
console.log(`API: Invalid quantity to add: ${quantityToAdd}. Must be a positive number.`);
return res.status(400).json({ message: 'Quantity to add must be a positive number.' });
}

try {
// It's generally safer to fetch the current quantity first, then update,
// or use a direct SQL increment if your DB supports it atomically and Drizzle allows it easily.
// For simplicity here, we'll use a SQL expression to increment.
// Note: Ensure your 'quantity' column can handle this operation (e.g., not have constraints violated).

```
const result = await db.update(plantsTable)
  .set({ quantity: sql`${plantsTable.quantity} + ${amount}` })
  .where(eq(plantsTable.id, id))
  .returning({ updatedId: plantsTable.id, newQuantity: plantsTable.quantity });

if (result.length === 0) {
  console.log(`API: Plant with ID ${id} not found for stock update.`);
  return res.status(404).json({ message: `Plant with ID ${id} not found.` });
}

const updatedPlant = result[0];
console.log(`API: Stock added successfully for plant ID ${updatedPlant.updatedId}. New quantity: ${updatedPlant.newQuantity}`);
return res.status(200).json({ 
  message: 'Stock added successfully', 
  plantId: updatedPlant.updatedId, 
  newQuantity: updatedPlant.newQuantity 
});
```

} catch (error: any) {
console.error(`API: Error adding stock for plant ID ${id}:`, error);
return res.status(500).json({ message: 'Failed to add stock.', error: error.message });
}
