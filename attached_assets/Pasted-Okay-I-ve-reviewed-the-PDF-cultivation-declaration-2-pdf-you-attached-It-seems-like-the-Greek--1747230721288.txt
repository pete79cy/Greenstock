const drawTextSafe = (
page: any,
text: any,
x: number,
y: number,

size: number,
maxWidth: number,
color = rgb(0, 0, 0)

let safeText = text === undefined || text === null ? '' : String(text);

// Truncate text if it's too wide for the cell
let textWidth = font.widthOfTextAtSize(safeText, size);
if (textWidth \> maxWidth && maxWidth \> 0) {
let truncatedText = '';
for (let i = safeText.length; i \> 0; i--) {
truncatedText = safeText.substring(0, i) + '...';
if (font.widthOfTextAtSize(truncatedText, size) \<= maxWidth) {
break;
}
}
safeText = truncatedText;
// Fallback if even "..." is too long (highly unlikely for reasonable maxWidth)
if (font.widthOfTextAtSize(safeText, size) \> maxWidth) {
safeText = safeText.substring(0, Math.floor(maxWidth / font.widthOfTextAtSize("...", size) \* 3)); // very rough estimate
}
}

page.drawText(safeText, { x, y, font, size, color });
};
