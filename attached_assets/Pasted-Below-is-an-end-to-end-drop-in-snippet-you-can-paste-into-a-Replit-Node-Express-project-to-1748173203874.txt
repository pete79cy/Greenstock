```bash
npm i puppeteer @handlebars/allow-prototype-access handlebars
```

## 3  Express route that streams the PDF

**`/server/routes/payslips.ts`**

```ts
import { Router } from "express";
import puppeteer from "puppeteer";
import * as fs from "node:fs/promises";
import path from "node:path";
import Handlebars from "handlebars";
import { allowInsecurePrototypeAccess } from "@handlebars/allow-prototype-access";

const hbs = allowInsecurePrototypeAccess(Handlebars as any);
const router = Router();

// GET  /api/payslips/:id/pdf
router.get("/:id/pdf", async (req, res, next) => {
  const id = Number(req.params.id);

  try {
    /* ---------------------------------------------
     * 1. Fetch payslip + employee data from DB
     *    (replace with your own ORM call)
     * ------------------------------------------- */
    const payslip = await db.payslip.findUnique({
      where: { id },
      include: { employee: true },
    });
    if (!payslip) return res.status(404).json({ message: "Not found" });

    /* ---------------------------------------------
     * 2. Prepare template context
     * ------------------------------------------- */
    const gross = payslip.grossSalary;
    const si = gross * 0.083;
    const gesy = gross * 0.0265;
    const net = gross - si - gesy;

    const view = {
      employee: {
        name: payslip.employee.fullName,
        designation: payslip.employee.position,
      },
      pay_period: payslip.payPeriod,                 // e.g. "2025-05"
      pay_date: payslip.payDate.toLocaleDateString("en-GB"),
      gross_salary_fmt: gross.toFixed(2),
      si_fmt: si.toFixed(2),
      gesy_fmt: gesy.toFixed(2),
      net_fmt: net.toFixed(2),
    };

    /* ---------------------------------------------
     * 3. Compile Handlebars template
     * ------------------------------------------- */
    const tplPath = path.join(__dirname, "..", "templates", "payslip.hbs");
    const templateSrc = await fs.readFile(tplPath, "utf8");
    const compile = hbs.compile(templateSrc);
    const html = compile(view);

    /* ---------------------------------------------
     * 4. Render to PDF with Puppeteer
     * ------------------------------------------- */
    const browser = await puppeteer.launch({
      args: ["--no-sandbox", "--disable-setuid-sandbox"],
    });
    const page = await browser.newPage();
    await page.setContent(html, { waitUntil: "networkidle0" });
    const pdf = await page.pdf({ format: "A4", printBackground: true });
    await browser.close();

    /* ---------------------------------------------
     * 5. Stream back to client
     * ------------------------------------------- */
    res.setHeader("Content-Type", "application/pdf");
    res.setHeader(
      "Content-Disposition",
      `inline; filename="payslip-${id}.pdf"`,
    );
    res.send(pdf);
  } catch (err) {
    next(err);
  }
});

export default router;
```

Then mount it in your main server file:

```ts
import payslipRouter from "./routes/payslips";
app.use("/api/payslips", payslipRouter);
```

## 4  Open it from the React side

```ts
// PayslipRow.tsx (button handler)
const openPdf = (id: number) => {
  window.open(`/api/payslips/${id}/pdf`, "_blank");
};
```

*(No `fetch()` needed — let the browser’s PDF viewer handle it.)*

---

### ✅  What this gives you

* **Clean HTML payslip** that prints beautifully on A4 or exports to PDF.
* **Single endpoint** (`/api/payslips/:id/pdf`) that compiles → renders → streams, so no temp files.
* Works on **Replit** (Puppeteer with `--no-sandbox`) and locally.

Plug it in, hit `/api/payslips/1/pdf`, and you should see the redesigned payslip in a new tab. If anything’s unclear, let me know and I’ll tweak the snippet.
