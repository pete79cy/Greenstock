A backup and restore function is an excellent idea for data safety and management!
Here's how we can implement this for your plant inventory:
 * Backup:
   * We'll create an API endpoint that fetches all your plant data from the database.
   * This data will be formatted as a JSON file.
   * When you trigger the backup, this JSON file will be downloaded to your computer.
 * Restore:
   * We'll create another API endpoint that allows you to upload a previously downloaded JSON backup file.
   * Important: The restore process will first clear all existing plant data from your database to prevent duplicates and ensure a clean restoration. Then, it will insert the data from your backup file. You'll be clearly informed about this before proceeding.
   * You'll get feedback on whether the restore was successful.
Let's start by creating the backend API endpoints for backup and restore.

Step 2: Create API Endpoint for Restore (restore.ts)
This will use multer for file uploads, which should already be installed from the previous "Excel Import" feature.
<lov-write file_path="src/pages/api/backup/restore.ts">
import type { NextApiRequest, NextApiResponse } from 'next';
import multer from 'multer';
import { db } from '@/server/db'; // Adjust path
import { plantsTable } from '@/server/db/schema'; // Adjust path
import { Plant } from '@/pages/api/plants'; // Assuming Plant type
// Configure multer for memory storage
const storage = multer.memoryStorage();
const upload = multer({ storage: storage });
// Helper to run multer middleware
const runMiddleware = (req: NextApiRequest & { file?: any }, res: NextApiResponse, fn: any) => {
return new Promise((resolve, reject) => {
fn(req, res, (result: any) => {
if (result instanceof Error) {
return reject(result);
}
return resolve(result);
});
});
};
export const config = {
api: {
bodyParser: false, // Disable Next.js body parsing for multer
},
};
export default async function handler(
req: NextApiRequest & { file?: any }, // Extend req type for multer's file
res: NextApiResponse<{ message: string; error?: any; restoredCount?: number }>
) {
if (req.method === 'POST') {
console.log("API: Initiating data restore...");
try {
await runMiddleware(req, res, upload.single('backupFile'));
  if (!req.file) {
    console.log("API: No backup file uploaded.");
    return res.status(400).json({ message: 'No backup file uploaded.' });
  }

  console.log(`API: Backup file received: ${req.file.originalname}, size: ${req.file.size} bytes`);

  const backupDataString = req.file.buffer.toString('utf-8');
  const plantsToRestore: Plant[] = JSON.parse(backupDataString);

  if (!Array.isArray(plantsToRestore)) {
    console.log("API: Invalid backup file format - expected an array of plants.");
    return res.status(400).json({ message: 'Invalid backup file format.' });
  }
  
  console.log(`API: Parsed ${plantsToRestore.length} plants from backup file.`);

  // Start a transaction
  await db.transaction(async (tx) => {
    console.log("API: Clearing existing plants data...");
    await tx.delete(plantsTable);
    console.log("API: Existing plants data cleared.");

    if (plantsToRestore.length > 0) {
      // Drizzle expects objects that match the table schema for insertion.
      // We need to remove any extra fields like `id` if the DB auto-generates it,
      // or ensure IDs are handled correctly based on schema (e.g., if they are part of backup).
      // For simplicity, assuming IDs in backup should be restored if present and schema allows.
      // If IDs are auto-incrementing and not part of backup or should be regenerated,
      // map the data to exclude `id`.
      const dataToInsert = plantsToRestore.map(p => ({
        name: p.name,
        scientificName: p.scientificName,
        plantingYear: p.plantingYear,
        quantity: p.quantity,
        // createdAt and updatedAt would typically be handled by DB defaults or triggers
        // If they are in the backup and you want to restore them, include them here.
        // For simplicity, we're relying on DB defaults or current timestamps if not.
      }));

      console.log(`API: Inserting ${dataToInsert.length} plants into the database...`);
      // @ts-ignore // Drizzle insert type might need specific casting if schema is complex
      await tx.insert(plantsTable).values(dataToInsert);
      console.log("API: Plants inserted successfully.");
    }
  });

  console.log("API: Data restore completed successfully.");
  return res.status(200).json({ message: 'Data restored successfully.', restoredCount: plantsToRestore.length });

} catch (error: any) {
  console.error('API: Error restoring data:', error);
  if (error instanceof SyntaxError) {
    return res.status(400).json({ message: 'Invalid JSON backup file.', error: error.message });
  }
  return res.status(500).json({ message: 'Failed to restore data.', error: error.message });
}

} else {
console.log(API: Method ${req.method} not allowed for /api/backup/restore);
res.setHeader('Allow', ['POST']);
res.status(405).end(Method ${req.method} Not Allowed);
}
}
